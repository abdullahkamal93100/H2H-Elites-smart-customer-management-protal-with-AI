from flask import Flask, jsonify, request
from flask_sqlalchemy import SQLAlchemy
from faker import Faker
import random
import datetime
import numpy as np
from sklearn.linear_model import LogisticRegression

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///crm.db'
app.config['SQLALCHEMY_TRACK_MODIFICATIONS'] = False

db = SQLAlchemy(app)
fake = Faker()

# -----------------------------
# DATABASE MODELS
# -----------------------------

class Customer(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(100))
    region = db.Column(db.String(50))
    plan = db.Column(db.String(50))
    contract_end = db.Column(db.Date)
    nps = db.Column(db.Integer)
    usage = db.Column(db.Float)

class Ticket(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    customer_id = db.Column(db.Integer)
    severity = db.Column(db.Integer)
    created_at = db.Column(db.DateTime, default=datetime.datetime.utcnow)

class Device(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    customer_id = db.Column(db.Integer)
    device_name = db.Column(db.String(100))

# -----------------------------
# SYNTHETIC DATA GENERATION
# -----------------------------

@app.route("/generate-data", methods=["POST"])
def generate_data():
    for _ in range(200):
        customer = Customer(
            name=fake.company(),
            region=random.choice(["APAC", "EU", "US"]),
            plan=random.choice(["Basic", "Pro", "Enterprise"]),
            contract_end=datetime.date.today() + datetime.timedelta(days=random.randint(30, 365)),
            nps=random.randint(-100, 100),
            usage=random.uniform(10, 1000)
        )
        db.session.add(customer)
        db.session.flush()

        for _ in range(random.randint(1, 5)):
            ticket = Ticket(
                customer_id=customer.id,
                severity=random.randint(1, 5)
            )
            db.session.add(ticket)

        for _ in range(random.randint(1, 3)):
            device = Device(
                customer_id=customer.id,
                device_name=fake.word()
            )
            db.session.add(device)

    db.session.commit()
    return jsonify({"message": "200 customers generated"})


# -----------------------------
# CRUD - CUSTOMER
# -----------------------------

@app.route("/customers", methods=["GET"])
def get_customers():
    customers = Customer.query.all()
    return jsonify([{
        "id": c.id,
        "name": c.name,
        "region": c.region,
        "plan": c.plan,
        "nps": c.nps
    } for c in customers])


@app.route("/customers", methods=["POST"])
def create_customer():
    data = request.json
    customer = Customer(**data)
    db.session.add(customer)
    db.session.commit()
    return jsonify({"message": "Customer created"})


# -----------------------------
# ACCOUNT HEALTH SCORE
# -----------------------------

def compute_health(customer):
    tickets = Ticket.query.filter_by(customer_id=customer.id).all()
    avg_severity = np.mean([t.severity for t in tickets]) if tickets else 1

    contract_days = (customer.contract_end - datetime.date.today()).days

    score = (
        0.4 * customer.nps +
        0.3 * (100 - avg_severity * 20) +
        0.3 * min(contract_days, 365)
    )

    return round(score, 2)


@app.route("/health/<int:customer_id>")
def health(customer_id):
    customer = Customer.query.get(customer_id)
    return jsonify({
        "customer": customer.name,
        "health_score": compute_health(customer)
    })


# -----------------------------
# CHURN PREDICTION
# -----------------------------

def train_churn_model():
    customers = Customer.query.all()

    X = []
    y = []

    for c in customers:
        tickets = Ticket.query.filter_by(customer_id=c.id).count()
        contract_days = (c.contract_end - datetime.date.today()).days

        X.append([c.nps, tickets, contract_days])
        y.append(1 if (c.nps < 0 or contract_days < 60) else 0)

    model = LogisticRegression()
    model.fit(X, y)

    return model


@app.route("/churn/<int:customer_id>")
def churn(customer_id):
    model = train_churn_model()
    c = Customer.query.get(customer_id)

    tickets = Ticket.query.filter_by(customer_id=c.id).count()
    contract_days = (c.contract_end - datetime.date.today()).days

    prob = model.predict_proba([[c.nps, tickets, contract_days]])[0][1]

    return jsonify({
        "customer": c.name,
        "churn_probability": float(prob),
        "risk": "HIGH" if prob > 0.6 else "LOW"
    })

@app.route("/query", methods=["POST"])
def nl_query():
    question = request.json.get("question")

    # Simple rule-based placeholder (replace with OpenAI later)
    if "high nps" in question.lower():
        customers = Customer.query.filter(Customer.nps > 50).all()
    else:
        customers = Customer.query.limit(5).all()

    return jsonify([c.name for c in customers])

@app.route("/email-summary/<int:customer_id>")
def email_summary(customer_id):
    c = Customer.query.get(customer_id)
    tickets = Ticket.query.filter_by(customer_id=c.id).count()

    summary = f"""
    Weekly Report for {c.name}:
    - Plan: {c.plan}
    - Region: {c.region}
    - NPS: {c.nps}
    - Tickets this period: {tickets}
    - Usage: {c.usage:.2f}

    Overall Health Score: {compute_health(c)}
    """

    return jsonify({"email": summary})
